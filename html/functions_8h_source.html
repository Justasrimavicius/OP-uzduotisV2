230 int i, j, k;
231 int n1 = m - l + 1;
232 int n2 = r - m;
233
234 T L(n1), R(n2);
235
236 for (i = 0; i < n1; i++)
237 L[i] = arr[l + i];
238 for (j = 0; j < n2; j++)
239 R[j] = arr[m + j + 1];
240
241 i = 0;
242 j = 0;
243 k = l;
244 while (i < n1 && j < n2){
245 if (L[i].getVardas() <= R[j].getVardas()){
246 arr[k] = L[i];
247 i++;
248 }

250 arr[k] = R[j];
251 j++;
252 }
